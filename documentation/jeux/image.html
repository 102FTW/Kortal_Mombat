class 
	IMAGE

inherit
	SDL_WRAPPER

create 
	make

feature {NONE} -- initialisation

	make (a_targetscreen: POINTER; a_path: STRING_8; a_x, a_y: INTEGER_32)
			-- Crée une image avec un path et une position x et y
		local
			path_image: C_STRING
			memory_manager: POINTER
		do
			screen := a_targetscreen
			create path_image.make (a_path)
			image := sdl_loadbmp (path_image.item)
			targetarea := memory_manager.memory_calloc (1, sdl_sizeof)
			set_x (a_x.to_integer_16.to_integer_32)
			set_y (a_y.to_integer_16.to_integer_32)
			set_w_rect (targetarea, get_w_image (image).to_natural_16)
			set_h_rect (targetarea, get_h_image (image).to_natural_16)
			setcolorkey
		end
	
feature 

	image: POINTER
			-- Contient l'adresse de l'image

	screen: POINTER
			-- Contient l'adresse de l'écran sur laquelle l'image doit etre affichée

	targetarea: POINTER
			--Contient les coordonées de l'image

	setcolorkey
			-- Set le couleur que SDL doit considérer comme invisible
		local
			error: INTEGER_32
		do
			error := sdl_setcolorkey (image, sdl_srccolorkey, sdl_maprgb (get_format (screen), 0.to_natural_8, 255.to_natural_8, 255.to_natural_8))
		end

	blitsurface
			-- Affiche l'image sur l'écran
		local
			error: INTEGER_32
		do
			error := sdl_blitsurface (image, create {POINTER}, screen, targetarea)
			check
				blitsurface: error = 0
			end
		end

	x: INTEGER_32 assign set_x
			-- Position x de l'image
		do
			Result := get_x_rect (targetarea).to_integer
		end

	y: INTEGER_32 assign set_y
			-- Position y de l'image
		do
			Result := get_y_rect (targetarea).to_integer
		end

	set_x (a_x: INTEGER_32)
			-- set la position x de l'image
		do
			set_x_rect (targetarea, a_x.to_integer_16)
		end

	set_y (a_y: INTEGER_32)
			-- set la position x de l'image
		do
			set_y_rect (targetarea, a_y.to_integer_16)
		end

	width: INTEGER_32
			-- contient la largeur de l'image
		do
			Result := get_w_image (image)
		end

	height: INTEGER_32
			-- contient la hauteur de l'image
		do
			Result := get_h_image (image)
		end
